Including them in the summary prompt gives the LLM visibility into what has been modified, by whom, and when -- producing far more useful document analysis. The cascading detection approach ensures the feature works across different Office versions, from the latest with full TrackedChange metadata (WordApi 1.6) down to the universal OOXML fallback (WordApi 1.1).

       Add OOXML parsing helpers (not exported, above extractTrackedChanges).
       These patterns are informed by the docx-redline-js reference library (reference/docx-redline-js/).

       ```javascript
       const W_NS = 'http://schemas.openxmlformats.org/wordprocessingml/2006/main';

       /**
        * Query OOXML elements with namespace-aware fallback.
        * Tries getElementsByTagNameNS first, falls back to prefix-based getElementsByTagName.
        * Pattern from docx-redline-js: browser DOMParser namespace resolution is inconsistent.
        */
       function queryElements(parent, localName) {
           let elements = parent.getElementsByTagNameNS(W_NS, localName);
           if (elements.length > 0) return Array.from(elements);
           elements = parent.getElementsByTagName('w:' + localName);
           return Array.from(elements);
       }

       function getChangeAuthor(element) {
           return element.getAttributeNS(W_NS, 'author')
               || element.getAttribute('w:author')
               || element.getAttribute('author')
               || null;
       }

       function getChangeDate(element) {
           return element.getAttributeNS(W_NS, 'date')
               || element.getAttribute('w:date')
               || element.getAttribute('date')
               || null;
       }

       /** Extract text from a run, handling w:t, w:delText, w:br, w:tab, w:cr. */
       function readRunText(run, useDelText = false) {
           let text = '';
           for (const child of Array.from(run.childNodes || [])) {
               if (child.nodeType !== 1) continue;
               const name = child.localName;
               if (name === 't' && !useDelText) text += child.textContent || '';
               else if (name === 'delText' && useDelText) text += child.textContent || '';
               else if (name === 'tab') text += '\t';
               else if (name === 'br' || name === 'cr') text += '\n';
           }
           return text;
       }

       /** Extract text from all runs within a revision element. */
       function extractRevisionText(element, useDelText = false) {
           const runs = queryElements(element, 'r');
           let text = '';
           for (const run of runs) text += readRunText(run, useDelText);
           if (!text && useDelText) {
               for (const dt of queryElements(element, 'delText')) text += dt.textContent || '';
           }
           return text;
       }

       /**
        * Get containing paragraph's current text for clause context.
        * Skips runs inside w:del or w:moveFrom containers.
        * Pattern from docx-redline-js ingestion-export.js.
        */
       function getContainingParagraphText(changeElement) {
           let node = changeElement;
           while (node && node.localName !== 'p') node = node.parentNode;
           if (!node) return '';
           const textNodes = queryElements(node, 't');
           let text = '';
           for (const t of textNodes) {
               let parent = t.parentNode, excluded = false;
               while (parent && parent !== node) {
                   if (parent.localName === 'del' || parent.localName === 'moveFrom') { excluded = true; break; }
                   parent = parent.parentNode;
               }
               if (!excluded) text += t.textContent || '';
           }
           return text.trim();
       }

       function getNextElementSibling(element) {
           let sibling = element.nextSibling;
           while (sibling && sibling.nodeType !== 1) sibling = sibling.nextSibling;
           return sibling;
       }

       /** Skip w:ins/w:del inside w:trPr (table row property markers, not content). */
       function isTableRowRevisionMarker(element) {
           const parent = element.parentNode;
           return parent && parent.localName === 'trPr';
       }

       /**
        * Parses OOXML for tracked changes using browser DOMParser.
        * Handles: w:ins, w:del, w:moveFrom, w:moveTo
        * Pairs adjacent w:del + w:ins (by DOM sibling, not array index) as replacements.
        * Extracts containing paragraph text for clause context.
        * Patterns informed by docx-redline-js reference library.
        */
       function parseOoxmlTrackedChanges(ooxml) {
           const changes = [];
           try {
               const parser = new DOMParser();
               const doc = parser.parseFromString(ooxml, 'application/xml');

               // Process deletions first, pairing with adjacent insertions via DOM siblings
               const deletions = queryElements(doc, 'del');
               const pairedInsertions = new Set();

               for (const del of deletions) {
                   if (isTableRowRevisionMarker(del)) continue;
                   const author = getChangeAuthor(del);
                   const date = getChangeDate(del);
                   const delText = extractRevisionText(del, true);
                   if (!delText.trim()) continue;
                   const paragraphText = getContainingParagraphText(del);

                   // DOM-based pairing: check if next element sibling is w:ins from same author
                   const nextSibling = getNextElementSibling(del);
                   if (nextSibling && nextSibling.localName === 'ins' &&
                       getChangeAuthor(nextSibling) === author) {
                       const insText = extractRevisionText(nextSibling, false);
                       if (insText.trim()) {
                           pairedInsertions.add(nextSibling);
                           changes.push({
                               type: 'Replaced', beforeText: delText, afterText: insText,
                               text: insText, author, date, paragraphText, tier: 3
                           });
                           continue;
                       }
                   }
                   changes.push({ type: 'Deleted', text: delText, author, date, paragraphText, tier: 3 });
               }

               // Process unpaired insertions
               for (const ins of queryElements(doc, 'ins')) {
                   if (pairedInsertions.has(ins) || isTableRowRevisionMarker(ins)) continue;
                   const insText = extractRevisionText(ins, false);
                   if (!insText.trim()) continue;
                   changes.push({
                       type: 'Added', text: insText, author: getChangeAuthor(ins),
                       date: getChangeDate(ins), paragraphText: getContainingParagraphText(ins), tier: 3
                   });
               }

               // Process move operations
               for (const mf of queryElements(doc, 'moveFrom')) {
                   const text = extractRevisionText(mf, true) || extractRevisionText(mf, false);
                   if (!text.trim()) continue;
                   changes.push({
                       type: 'Moved (from)', text, author: getChangeAuthor(mf),
                       date: getChangeDate(mf), paragraphText: getContainingParagraphText(mf), tier: 3
                   });
               }
               for (const mt of queryElements(doc, 'moveTo')) {
                   const text = extractRevisionText(mt, false);
                   if (!text.trim()) continue;
                   changes.push({
                       type: 'Moved (to)', text, author: getChangeAuthor(mt),
                       date: getChangeDate(mt), paragraphText: getContainingParagraphText(mt), tier: 3
                   });
               }
           } catch (e) {
               console.error('Failed to parse OOXML for tracked changes:', e);
           }
           return changes;
       }
       ```

       Add the main function (exported):
       ```javascript
       /**
        * Extracts tracked changes from the active document using cascading API detection.
        *
        * Tier 1 (WordApi 1.6): body.getTrackedChanges() - full metadata (author, date, type, text)
        * Tier 2 (WordApi 1.4): body.getReviewedText() - plain text with/without changes applied
        * Tier 3 (WordApi 1.1): body.getOoxml() - parse OOXML revision marks via DOMParser
        *
        * Detection uses Office.context.requirements.isSetSupported() at runtime,
        * following the project's existing pattern for WordApi 1.4 detection.
        *
        * @returns {Promise<{tier: number, changes: Array<{type: string, text: string, author: string|null, date: string|null, tier: number}>}>}
        */
       export async function extractTrackedChanges() {
           // Determine the best available API tier
           let tier = 3; // Default to OOXML fallback
           try {
               if (typeof Office !== 'undefined' && Office.context && Office.context.requirements) {
                   if (Office.context.requirements.isSetSupported('WordApi', '1.6')) {
                       tier = 1;
                   } else if (Office.context.requirements.isSetSupported('WordApi', '1.4')) {
                       tier = 2;
                   }
               }
           } catch (e) {
               // Office detection failed, use tier 3 fallback
           }

           try {
               if (tier === 1) {
                   return await extractTrackedChangesTier1();
               } else if (tier === 2) {
                   return await extractTrackedChangesTier2();
               } else {
                   return await extractTrackedChangesTier3();
               }
           } catch (e) {
               console.error('Failed to extract tracked changes:', e);
               return { tier: 0, changes: [] };
           }
       }
       ```

       Add the three tier-specific functions (not exported, internal helpers):
       ```javascript
       /**
        * Tier 1: WordApi 1.6 - Full tracked change metadata with paragraph context.
        * Loads the containing paragraph text for each change so the LLM sees the full clause.
        */
       async function extractTrackedChangesTier1() {
           const changes = [];
           await Word.run(async (context) => {
               const trackedChanges = context.document.body.getTrackedChanges();
               trackedChanges.load('items');
               await context.sync();

               for (const tc of trackedChanges.items) {
                   tc.load('author,date,text,type');
               }
               await context.sync();

               // Load paragraph context for each non-None change
               const rangeData = [];
               for (const tc of trackedChanges.items) {
                   if (tc.type === 'None') continue;
                   const range = tc.getRange();
                   const para = range.paragraphs.getFirst();
                   para.load('text');
                   rangeData.push({ tc, para });
               }
               if (rangeData.length > 0) {
                   await context.sync();
               }

               for (const { tc, para } of rangeData) {
                   changes.push({
                       type: tc.type,
                       text: tc.text,
                       author: tc.author,
                       date: tc.date,
                       paragraphText: para.text || '',
                       tier: 1
                   });
               }
           });
           return { tier: 1, changes };
       }

       /**
        * Tier 2: WordApi 1.4 - Reviewed text (current vs original).
        * No individual change metadata, but shows the net effect of all changes.
        */
       async function extractTrackedChangesTier2() {
           const changes = [];
           await Word.run(async (context) => {
               const body = context.document.body;
               const currentText = body.getReviewedText('Current');
               const originalText = body.getReviewedText('Original');
               await context.sync();

               changes.push({
                   type: 'reviewed',
                   text: currentText.value,
                   author: null,
                   date: null,
                   tier: 2
               });
               changes.push({
                   type: 'original',
                   text: originalText.value,
                   author: null,
                   date: null,
                   tier: 2
               });
           });
           return { tier: 2, changes };
       }

       /**
        * Tier 3: WordApi 1.1 - OOXML parsing for revision marks.
        * Uses browser DOMParser to extract w:ins and w:del elements.
        */
       async function extractTrackedChangesTier3() {
           let ooxml = '';
           await Word.run(async (context) => {
               const body = context.document.body;
               const result = body.getOoxml();
               await context.sync();
               ooxml = result.value;
           });
           const changes = parseOoxmlTrackedChanges(ooxml);
           return { tier: 3, changes };
       }
       ```
