Wire the unified LLM client into the UI with backend selector, per-backend config, and config migration.

```javascript
// src/lib/llm-client.js — created by Plan 01

/**
 * Strips <think>...</think> tags and reasoning artifacts from LLM responses.
 * @param {string} text - Raw LLM response text
 * @param {function} [log] - Optional logging function (called with message, type)
 * @returns {string} Cleaned text
 */
export function stripThinkTags(text, log);

/**
 * Sends a prompt to the configured LLM backend via OpenAI-compatible API.
 * @param {object} config - { url: string, apiKey: string, model: string }
 * @param {string} promptText - The full prompt text (with {selection} already replaced)
 * @param {function} [log] - Optional logging function
 * @returns {Promise<string>} The LLM response text (think tags stripped)
 */
export async function sendPrompt(config, promptText, log);

/**
 * Tests connection and retrieves model information via /v1/models.
 * @param {object} config - { url: string, apiKey: string }
 * @returns {Promise<{ connected: boolean, models: Array<{id: string}> }>}
 */
export async function testConnection(config);
```
